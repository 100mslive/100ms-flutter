static HMSCodec getHMSCodecFromName(String name) {
  switch (name) {
    case 'kHMSCodecH264':
      return HMSCodec.kHMSCodecH264;
    case 'kHMSCodecVP8':
      return HMSCodec.kHMSCodecVP8;

    default:
      return HMSCodec.unknown;
  }
}
